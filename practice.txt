#include<stduio.h>
#include<iostream>
using namespace std

int main()
{
    int i,j,n,temp; //
//
    cin>>n;
    int arr[n];

    //Loop Taking input//
    for(i=0;i<n;i++)
      cin>>arr[i];   //
    
    //Sorting in ascending order
    for(i=0;i<n;i++)
    {
        for(j=i+1;j<n;j++)
        {
            if(arr[i]>arr[j])
            {
                temp = arr[i];//
                arr[i] = arr[j];
                arr[j] = temp; 
            }
        }
    }

    for(i=0;i<n;i++) //
       cout<<" "<<arr[i];
    
    return 0;
}
